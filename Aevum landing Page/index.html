# Initialize AEVUM client
from aevum import AEVUM

client = AEVUM(api_key="aev_...")

# Analyse a document
result = client.analyse(
    file="Q3_Report.pdf",
    tasks=["summarize", "action_items"],
    output="notion:workspace/Q3"
)

# Draft a response
draft = client.write(
    type="email",
    context=result,
    tone="executive",
    variants=3
)

print(draft.best)
